  <script>

    const canvas = document.getElementById("canvas");

    const ctx = canvas.getContext("2d");

    const controls = {

      angle: document.getElementById("angleControl"),

      depth: document.getElementById("depthControl"),

      length: document.getElementById("lengthControl"),

      reduction: document.getElementById("reductionControl"),

    };

    let audioCtx = new (window.AudioContext || window.webkitAudioContext)();

    let osc, gain, lfo;

    const modulation = {

      angle: { active: false, lastTime: 0 },

      depth: { active: false, lastTime: 0 },

      length: { active: false, lastTime: 0 },

      reduction: { active: false, lastTime: 0 },

    };

    function drawTree(x, y, len, angle, depth, maxDepth, reduction) {

      if (depth > maxDepth) return;

      const x2 = x + len * Math.cos(angle);

      const y2 = y - len * Math.sin(angle);

      ctx.beginPath();

      ctx.moveTo(x, y);

      ctx.lineTo(x2, y2);

      ctx.stroke();

      drawTree(x2, y2, len * reduction, angle - parseFloat(controls.angle.value), depth + 1, maxDepth, reduction);

      drawTree(x2, y2, len * reduction, angle + parseFloat(controls.angle.value), depth + 1, maxDepth, reduction);

    }

    function startFractal() {

      ctx.clearRect(0, 0, canvas.width, canvas.height);

      ctx.strokeStyle = "lime";

      const depth = parseInt(controls.depth.value);

      const length = parseInt(controls.length.value);

      const reduction = parseFloat(controls.reduction.value);

      drawTree(canvas.width / 2, canvas.height, length, Math.PI / 2, 0, depth, reduction);

      const freq = 100 + length * 2 + depth * 20;

      playPad(freq, depth);

    }

    function playPad(freq, depth) {

      stopSound();

      osc = audioCtx.createOscillator();

      gain = audioCtx.createGain();

      lfo = audioCtx.createOscillator();

      let lfoGain = audioCtx.createGain();

      osc.type = "sawtooth";

      osc.frequency.value = freq;

      gain.gain.value = 0.1 + depth * 0.01;

      lfo.frequency.value = 1 + depth * 0.2;

      lfoGain.gain.value = 0.05;

      lfo.connect(lfoGain).connect(gain.gain);

      osc.connect(gain).connect(audioCtx.destination);

      osc.start();

      lfo.start();

    }

    function stopSound() {

      if (osc) osc.stop();

      if (lfo) lfo.stop();

    }

    function toggleMod(param) {

      modulation[param].active = !modulation[param].active;

    }

    function updateModulation(time) {

      Object.keys(modulation).forEach(param => {

        if (!modulation[param].active) return;

        const now = performance.now();

        const rate = parseInt(document.getElementById(`rate_${param}`).value);

        if (now - modulation[param].lastTime >= rate) {

          let ctrl = controls[param];

          let min = parseFloat(ctrl.min);

          let max = parseFloat(ctrl.max);

          let step = parseFloat(ctrl.step) || 1;

          let val = parseFloat(ctrl.value);

          const isRand = document.getElementById(`rand_${param}`).checked;

          if (isRand) {

            val = min + Math.random() * (max - min);

          } else {

            val += step;

            if (val > max) val = min;

          }

          ctrl.value = val.toFixed(step < 1 ? 2 : 0);

          modulation[param].lastTime = now;

        }

      });

    }

    function animate() {

      updateModulation();

      startFractal();

      requestAnimationFrame(animate);

    }

    animate();

    Object.values(controls).forEach(c => c.oninput = () => startFractal());
